### 历史对话压缩问题
**用户**：历史对话压缩有啥用，保存在哪里了吗

**助手**：历史对话压缩是一种优化技术，主要有以下作用：

1. **减少上下文长度**：避免对话内容过长导致超出模型的上下文窗口限制，确保系统能处理更长的对话历史。

2. **提高处理效率**：压缩后的对话占用更少的内存和计算资源，使系统响应更快。

3. **保持对话连贯性**：通过保留关键信息和核心内容，确保对话的逻辑连续性，同时去除冗余信息。

4. **优化用户体验**：使系统能够更好地理解用户的长期需求和对话意图。

关于存储位置：
- 压缩后的对话摘要会作为上下文的一部分保留在系统的临时存储中
- 具体的存储机制因系统实现而异，但通常是在内存中临时保存
- 历史对话的原始内容不会被永久存储，只有压缩后的关键信息会被保留
- 当对话结束或系统重启时，这些临时存储的信息会被清除

# 修复后的 get_history 函数
def get_history():
    """获取历史记录"""
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()
    
    # 获取匿名ID
    anonymous_id = request.cookies.get('anonymous_id')
    
    # 查询 core_function_results 表中的新记录
    cursor.execute('''
    SELECT id, anonymous_id, task_aversion, result_value, self_control, w1, w2, w3, score, risk_level, 
           model_predicted_level, suggestion, timestamp, user_notes
    FROM core_function_results 
    WHERE anonymous_id = ?
    ORDER BY timestamp DESC
    ''', (anonymous_id,))
    
    new_records = cursor.fetchall()
    
    # 查询 risk_records 表中的旧记录
    cursor.execute('''
    SELECT id, anonymous_id, task_aversion, result_value, self_control, 
           0.4 as w1, 0.3 as w2, 0.3 as w3, 
           (task_aversion * 0.4 + (100 - result_value) * 0.3 + (100 - self_control) * 0.3) as score, 
           CASE 
               WHEN (task_aversion * 0.4 + (100 - result_value) * 0.3 + (100 - self_control) * 0.3) > 70 THEN '高'
               WHEN (task_aversion * 0.4 + (100 - result_value) * 0.3 + (100 - self_control) * 0.3) > 40 THEN '中'
               ELSE '低'
           END as risk_level, 
           NULL as model_prediction, '一致' as comparison, 
           CASE 
               WHEN (task_aversion * 0.4 + (100 - result_value) * 0.3 + (100 - self_control) * 0.3) > 70 THEN '建议你立即采取行动，分解任务并设定明确的截止日期，避免拖延带来的严重后果。'
               WHEN (task_aversion * 0.4 + (100 - result_value) * 0.3 + (100 - self_control) * 0.3) > 40 THEN '建议你制定合理的计划，使用番茄工作法等时间管理技巧，逐步提高工作效率。'
               ELSE '你的拖延风险较低，继续保持良好的工作习惯，定期反思和调整工作方法。'
           END as suggestion, 
           timestamp, NULL as user_notes
    FROM risk_records 
    WHERE anonymous_id = ?
    ORDER BY timestamp DESC
    ''', (anonymous_id,))
    
    old_records = cursor.fetchall()
    conn.close()
    
    # 合并新旧记录
    all_records = new_records + old_records
    
    # 按时间戳排序
    all_records.sort(key=lambda x: x[13], reverse=True)
    
    history = []
    for record in all_records:
        history.append({
            'id': record[0],
            'task_aversion': record[2],
            'result_value': record[3],

            'risk_level': record[9],
            'model_predicted_level': record[10],
            'suggestion': record[11],
            'user_notes': record[12],
            'timestamp': record[13]
        })
    
    return jsonify({
        'status': 'success',

    })
